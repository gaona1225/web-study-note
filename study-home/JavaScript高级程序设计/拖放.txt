<!doctype html>
<html>
<head>
<meta charset="utf-8">
<title>拖放</title>
<style>
	img{height:100px;width:100px;}
</style>
<script>
	function handleMouseMove(event){
		if(event.button == 1){
			event.srcElement.dragDrop();
		}
	}

	function handleDragDropEvent(event){
		event.dataTransfer.setData('text',"http://www.baidu.com");
	}
</script>

</head>

<body>
	<form>
		<p>Try dragging the image.</p>
		<p>

		</p>
